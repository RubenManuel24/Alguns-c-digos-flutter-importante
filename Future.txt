import 'dart:html';

import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;
import 'dart:convert';
import 'dart:async';
class Home extends StatefulWidget {
  Home({Key? key}) : super(key: key);

  @override
  State<Home> createState() => _HomeState();
}

class _HomeState extends State<Home> {

  Future<Map> _carregarPreco() async{

    final urlcarregar =  Uri.https(
      "//blockchain.info",
      "/ticker");

    final Response = await http.get(urlcarregar);
        return json.decode(Response.body);
  }
  @override
  Widget build(BuildContext context) {
    return FutureBuilder(
      future: _carregarPreco(),
      builder: 
      (context,AsyncSnapshot<Map> snapshot) {
         String resultado = "";
        switch(snapshot.connectionState){
          case ConnectionState.none: 
             print("conexao none");
             break;
          case ConnectionState.active: 
             print("conexao ativa");
             break;
          case ConnectionState.done: 
             print("conexao done");
             if(snapshot.hasError){
               resultado = "Erro ao carregar os dados.";
             }
             else{
              double valor = snapshot.data!["USD"]["buy"];
              resultado = "Pre√ßo do bitcoin: ${valor.toString()}";
             }
             break;
          case ConnectionState.waiting: 
             print("conexao waite");
             resultado = "Carregando...";
             break;
        }

        return Center(
          child: Text(resultado),
        );
        },);
  }
}
